import { Steps, Tabs, TabItem } from '@astrojs/starlight/components';

<Tabs>
   <TabItem label="CLI" icon="setting">
      <Steps>

         1. Install

            
            ```console
            pnpm i -D @reuters-graphics/graphics-kit-publisher
            ```

         2. Add a config file to the root of your project. See [config options](/graphics-kit-publisher/config/).

            ```typescript
            // publisher.config.ts
            import { defineConfig } from '@reuters-graphics/graphics-kit-publisher';

            export default defineConfig({
            // Config options
            });
            ```

         3. Publish a preview of your project.


            ```console
            pnpm graphics-publisher preview
            ```


         4. Upload your project to the graphics server.

            ```console
            pnpm graphics-publisher upload
            ```

         5. Publish your project.

            ```console
            pnpm graphics-publisher publish
            ```

         6. Go to the pub. üç∫
         </Steps>
   </TabItem>
   <TabItem label="Graphics kit" icon="laptop">
      <Steps>

         1. In the graphics kit, this library should already be installed. If for some reason it's not, run:

            ```console
            pnpm i -D @reuters-graphics/graphics-kit-publisher
            ```

         2. You should already have a config file to the root of your project. See [config options](/graphics-kit-publisher/config/).

            ```typescript
            // publisher.config.ts
            import { defineConfig } from '@reuters-graphics/graphics-kit-publisher';

            export default defineConfig({
            // Config options
            });
            ```

         3. Publish a preview of your project.


            ```console
            pnpm preview
            ```


         4. Upload your project to the graphics server.

            ```console
            pnpm upload
            ```

         5. Publish your project.

            ```console
            pnpm pub
            ```

         6. Go to the pub. üç∫
         </Steps>
   </TabItem>
